Plugin structure

package.json:

{
    "name": "exampleplugin",
    "main": "exampleplugin.js",
    "version": "1.0.0",
    "displayName": "Example plugin",
    "author": "Me",
    "description": "Example plugin to test GreenFrog API"
}

code:

module.exports = {
  name: "ExamplePlugin",
  version: "1.0", 

  onLoad() {
    Logger.log(`${this.name} > Hello world!`);
  },

  onShutdown() {
    Logger.log(`${this.name} > Good bye!`);
  },
};
